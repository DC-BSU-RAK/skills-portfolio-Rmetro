import tkinter as tk
import random

class JokeApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Joke Assistant")
        self.root.geometry("600x480")
        self.root.resizable(False, False)

        # Background Canvas
        self.canvas = tk.Canvas(root, width=600, height=480, highlightthickness=0)
        self.canvas.place(x=0, y=0)

        # Colors for animated gradient
        self.color1 = [66, 135, 245]    
        self.animate_gradient()

        # frame for text
        self.text_frame = tk.Frame(root, bg="white", bd=3, relief="groove")
        self.text_frame.place(relx=0.5, rely=0.45, anchor="center", width=500, height=220)

        # Title 
        self.title = tk.Label(
            root,
            text="Alexa Joke Assistant",
            font=("Arial", 22, "bold"),
            bg="#1e1e1e",
            fg="white",
            padx=10,
            pady=5
        )
        self.title.place(relx=0.5, y=40, anchor="center")

        # jokes setup 
        self.setup_label = tk.Label(
            self.text_frame,
            text="",
            font=("Arial", 14),
            wraplength=480,
            bg="white",
            justify="left"
        )
        self.setup_label.pack(pady=10)

        # Joke punchline 
        self.punchline_label = tk.Label(
            self.text_frame,
            text="",
            font=("Arial", 12, "italic"),
            fg="#1E3A8A",
            wraplength=480,
            bg="white"
        )
        self.punchline_label.pack()

        # Buttons frame
        btn_frame = tk.Frame(root, bg="#1e1e1e")
        btn_frame.place(relx=0.5, rely=0.82, anchor="center")

        self.create_button(btn_frame, "Alexa tell me a Joke", self.show_setup).grid(row=0, column=0, padx=10, pady=5)
        self.create_button(btn_frame, "Show Punchline", self.show_punchline).grid(row=0, column=1, padx=10, pady=5)
        self.create_button(btn_frame, "Next Joke", self.show_setup).grid(row=1, column=0, padx=10, pady=5)
        self.create_button(btn_frame, "Quit", self.root.quit, bg="#E11D48").grid(row=1, column=1, padx=10, pady=5)

            "Why did the can crusher quit his job?Because it was soda pressing.",

            "What does a house wear?A dress.",
            "Why couldn't the toilet paper cross the road?Because it got stuck in a crack.",
            "Why didn't the bike want to go anywhere?Because it was two-tired!",
            "Want to hear a pizza joke?Nahhh, it's too cheesy!",
            "Why are chemists great at solving problems?Because they have all of the solutions!",
            "Why is it impossible to starve in the desert?Because of all the sand which is there!",
            "What did the cheese say when it looked in the mirror?Halloumi!",
            "Why did the developer go broke?Because he used up all his cache.",
            "Did you know that ants are the only animals that don't get sick?They have little antibodies.",
            "Why did the donut go to the dentist?To get a filling.",
            "What do you call a bear with no teeth?A gummy bear!",
            "What does a vegan zombie like to eat?Graaains.",
            "What do you call a dinosaur with only one eye?A Do-you-think-he-saw-us!",
            "Why should you never fall in love with a tennis player?Because to them... love means NOTHING!",
            "What did the full glass say to the empty glass?You look drunk.",
            "What's a potato's favorite form of transportation?The gravy train",

        self.current_setup = ""

    def create_button(self, parent, text, command, bg="#2563EB"):
        return tk.Button(
            parent,
            text=text,
            command=command,
            width=18,
            font=("Arial", 11, "bold"),
            bg=bg,
            fg="white",
            activebackground="#1E40AF",
            activeforeground="white",
            relief="raised",
            bd=3
        )

    def show_setup(self):
        joke = random.choice(self.jokes)
        setup, punchline = joke.split("?", 1)
        setup += "?"
        self.current_setup = setup
        self.current_punchline = punchline

        self.setup_label.config(text=setup)
        self.punchline_label.config(text="")

    def show_punchline(self):
        self.punchline_label.config(text=self.current_punchline)

    def animate_gradient(self):
        # Animate background color 
        for i in range(3):
            self.color1[i] = (self.color1[i] + 1) % 255

        color_hex = "#%02x%02x%02x" % tuple(self.color1)
        self.canvas.create_rectangle(0, 0, 600, 480, fill=color_hex, outline="")

        self.root.after(60, self.animate_gradient)


# Run code
root = tk.Tk()
app = JokeApp(root)
root.mainloop()
